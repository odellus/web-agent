using Reactant
using MLIR

@reactant function vecadd(a, b)
    c = similar(a)
    @inbounds for i in eachindex(a)
        c[i] = a[i] + b[i]
    end
    return c
end

# Compile to MLIR and run
a = rand(Float32, 1000)
b = rand(Float32, 1000)
compiled = lower_to_mlir(vecadd)
result = compiled(a, b)
